PASSWORD_TEST_KEY = "ETC_MIN_PASSWORD_LEN"
FULL_CONNECT_KEY = "SYS_VERSION"
ESSENTIAL_KEYS = ("SYS_VERSION", "SYS_MODELNAME", "SYS_BOARDID")


@dataclass(frozen=True)
class PasswordChangeResult:
    ok: bool
    used_password: str
    changed_to: str
    detail: str | None = None


@dataclass(frozen=True)
class AttemptResult:
    ok: bool
    base_url: str
    root_path: str
    auth_scheme: str
    flavor: str

    sys_version: str | None = None
    effective_username: str | None = None
    effective_password: str | None = None

    recovered: bool = False
    default_password_state: bool | None = None
    password_changed: bool | None = None

    error: AppError | None = None

    def to_phase1_response(self) -> Phase1Response:
        return Phase1Response(
            ok=self.ok,
            base_url=self.base_url,
            root_path=self.root_path,
            auth_scheme=self.auth_scheme,
            flavor=self.flavor,
            sys_version=self.sys_version,
            effective_username=self.effective_username,
            effective_password=self.effective_password,
            recovered=self.recovered,
            default_password_state=self.default_password_state,
            password_changed=self.password_changed,
            error=self.error,
        )


def tencryption_js(text: str) -> str:
    raw = text or ""
    return base64.b64encode(raw.encode("utf-8")).decode("ascii")


def tencode_url_js(b64_str: str) -> str:
    if not b64_str:
        return ""

    converted = b64_str.replace("+", "-").replace("/", "_").rstrip("=")
    if not converted:
        return ""

    first = b64_str[len(converted) - 1]
    second = b64_str[len(converted) // 2]
    return first + second + converted


def _unique_keep_order(items: list[str | None]) -> list[str]:
    out: list[str] = []
    for item in items:
        value = (item or "").strip()
        if not value:
            continue
        if value not in out:
            out.append(value)
    return out


def _resolve_target_password(req: Phase1Request, settings: AppSettings) -> str:
    """
    legacy/basic/TTA 쪽 target password는 사용자 선택값을 우선한다.
    지정이 없으면 기본 펌웨어 운영 비밀번호(123)를 기본값으로 사용한다.
    """
    value = (req.target_password or "").strip()
    if value:
        return value

    if getattr(settings, "default_password", ""):
        return settings.default_password

    return "123"


def _resolve_sec3_username(req: Phase1Request, settings: AppSettings) -> str:
    value = (req.sec3_username or "").strip()
    if value:
        return value
    return getattr(settings, "sec3_username", "") or "TruenTest"


def _resolve_sec3_password(req: Phase1Request, settings: AppSettings) -> str:
    value = (req.sec3_password or "").strip()
    if not value:
        value = getattr(settings, "sec3_password", "") or "!Camera1108"

    # Security 3.0 운영 비밀번호는 대문자 C 기준으로 정규화
    if value.lower() == "!camera1108":
        return "!Camera1108"
    return value


def _is_factory_reset_password(password: str) -> bool:
    value = (password or "").strip()
    return value in ("1234", "admin")


def _candidate_passwords(req: Phase1Request, settings: AppSettings) -> tuple[list[str], bool]:
    """
    정책:
    1) 입력 비밀번호가 1234 또는 admin이면:
       -> factory reset 상태로 보고 1234/admin 경로로 먼저 평가
       -> 필요 시 target password 변경 로직 진입
    2) 입력 비밀번호가 123 / !camera1108 / !Camera1108 등이면:
       -> 이미 변경된 제품에 직접 접속으로 보고 그 입력값만 사용
       -> 자동 비밀번호 변경 로직 실행 안 함
    """
    raw_password = (req.password or "").strip()

    if _is_factory_reset_password(raw_password):
        if raw_password == "admin":
            return ["admin", "1234"], True
        return ["1234", "admin"], True

    if raw_password:
        return [raw_password], False

    # 사용자가 비밀번호를 비워서 호출한 경우만 최소 fallback 허용
    fallback = _unique_keep_order(
        [
            "1234",
            "admin",
            getattr(settings, "default_password", "") or "123",
            getattr(settings, "tta_password", "") or "!camera1108",
            "!Camera1108",
        ]
    )
    return fallback, False


def _read_essentials(cli: CameraHttpClient) -> dict[str, str]:
    out: dict[str, str] = {}
    for key in ESSENTIAL_KEYS:
        out[key] = cli.read_param_value(key)
    return out


def _is_transport_miss(err: AppError | None) -> bool:
    if err is None:
        return False

    if err.kind in ("ssl", "timeout", "network"):
        return True

    detail = (err.detail or "").lower()
    return (
        "wrong version number" in detail
        or "remote end closed connection" in detail
        or "remotedisconnected" in detail
        or "connection refused" in detail
        or "timed out" in detail
        or "read timed out" in detail
        or "connection reset by peer" in detail
    )


def _is_default_password_state(err: AppError | None) -> bool:
    if err is None:
        return False

    status = err.status_code
    detail = (err.detail or "").lower()
    message = (err.message or "").lower()

    if status == 400:
        return True

    if "bad request" in detail or "bad request" in message:
        return True

    if status in (401, 403) and any(
        token in detail
        for token in (
            "document error",
            "access error",
            "parse error",
            "policy",
            "access denied",
        )
    ):
        return True

    return False


def _build_client(
    *,
    base_url: str,
    root_path: str,
    username: str,
    password: str,
    auth_scheme: str,
    verify_tls: bool,
    timeout_sec: float,
) -> CameraHttpClient:
    return CameraHttpClient(
        base_url=base_url,
        root_path=root_path,
        username=username,
        password=password,
        auth_scheme=auth_scheme,
        timeout_sec=timeout_sec,
        verify_tls=verify_tls,
    )


def _verify_after_password_change(
    *,
    base_url: str,
    root_path: str,
    username: str,
    new_password: str,
    auth_scheme: str,
    verify_tls: bool,
    timeout_sec: float,
) -> dict[str, str]:
    cli = _build_client(
        base_url=base_url,
        root_path=root_path,
        username=username,
        password=new_password,
        auth_scheme=auth_scheme,
        verify_tls=verify_tls,
        timeout_sec=timeout_sec,
    )

    last_err: AppError | None = None

    for _ in range(8):
        try:
            _ = cli.read_param_text(PASSWORD_TEST_KEY)
            return _read_essentials(cli)
        except AppError as exc:
            last_err = exc
            time.sleep(0.5)

    raise last_err or AppError(kind="auth", message="verify failed after password change")


def _write_usr_modpass(cli: CameraHttpClient, username: str, enc_old: str, enc_new: str) -> str:
    from urllib.parse import quote

    msg = (
        "USR_MODPASS="
        + quote(username, safe="")
        + ":"
        + quote(enc_old, safe="")
        + ":"
        + quote(enc_new, safe="")
    )

    tail = f"WriteParam?action=writeparam&{msg}"
    resp = cli.request_tail(tail)

    if resp.status == 200:
        return (resp.body or "").strip()

    if resp.status in (401, 403):
        raise AppError(
            kind="auth",
            message="authentication failed",
            status_code=resp.status,
            detail=(resp.body or "")[:300],
        )

    raise AppError(
        kind="http",
        message="password change rejected",
        status_code=resp.status,
        detail=(resp.body or "")[:300],
    )


def change_password_if_needed(
    *,
    base_url: str,
    root_path: str,
    username: str,
    old_password: str,
    new_password: str,
    timeout_sec: float,
    verify_tls: bool,
    auth_scheme: str,
) -> None:
    cli_old = _build_client(
        base_url=base_url,
        root_path=root_path,
        username=username,
        password=old_password,
        auth_scheme=auth_scheme,
        verify_tls=verify_tls,
        timeout_sec=timeout_sec,
    )

    crypto_type = 0
    public_key_pem = ""

    try:
        txt = cli_old.read_param_text("SYS_PUBLIC_KEY_CRYPTO")
        if "SYS_PUBLIC_KEY_CRYPTO=1" in (txt or ""):
            crypto_type = 1
    except Exception:
        crypto_type = 0

    if crypto_type == 1:
        try:
            txt2 = cli_old.read_param_text("SYS_PUBLIC_KEY")
            from infra.network.security3 import normalize_public_key_pem
            public_key_pem = normalize_public_key_pem(txt2)
        except Exception:
            public_key_pem = ""

    # RSA 우선
    if crypto_type == 1 and public_key_pem:
        try:
            enc_old = rsa_encrypt_with_pem(public_key_pem, old_password)
            enc_new = rsa_encrypt_with_pem(public_key_pem, new_password)
            body = _write_usr_modpass(cli_old, username, enc_old, enc_new)
            if not body or body.lower().startswith("ok"):
                _verify_after_password_change(
                    base_url=base_url,
                    root_path=root_path,
                    username=username,
                    new_password=new_password,
                    auth_scheme=auth_scheme,
                    verify_tls=verify_tls,
                    timeout_sec=timeout_sec,
                )
                return
        except Exception as exc:
            log.info("[MODPASS] RSA path failed, fallback to TENC: %s", exc)

    # TENC fallback
    enc_old = tencode_url_js(tencryption_js(old_password))
    enc_new = tencode_url_js(tencryption_js(new_password))
    body = _write_usr_modpass(cli_old, username, enc_old, enc_new)

    if body and not body.lower().startswith("ok"):
        raise AppError(kind="http", message="password change rejected", detail=body[:300])

    _verify_after_password_change(
        base_url=base_url,
        root_path=root_path,
        username=username,
        new_password=new_password,
        auth_scheme=auth_scheme,
        verify_tls=verify_tls,
        timeout_sec=timeout_sec,
    )


def try_recover_password(
    *,
    base_url: str,
    root_path: str,
    username: str,
    password_candidates: list[str],
    target_password: str,
    timeout_sec: float,
    verify_tls: bool,
    auth_scheme: str,
) -> PasswordChangeResult:
    last_err: AppError | None = None

    for old_pw in password_candidates:
        try:
            change_password_if_needed(
                base_url=base_url,
                root_path=root_path,
                username=username,
                old_password=old_pw,
                new_password=target_password,
                timeout_sec=timeout_sec,
                verify_tls=verify_tls,
                auth_scheme=auth_scheme,
            )
            return PasswordChangeResult(ok=True, used_password=old_pw, changed_to=target_password)

        except AppError as exc:
            last_err = exc
            continue
        except Exception as exc:
            last_err = AppError(kind="network", message="password recover failed", detail=str(exc))
            continue

    raise last_err or AppError(kind="auth", message="password recover failed")


def _attempt_candidate_combo(
    *,
    req: Phase1Request,
    base_url: str,
    root_path: str,
    auth_scheme: str,
    flavor: str,
    password_candidates: list[str],
    target_password: str,
    allow_recovery: bool,
    verify_tls: bool,
    timeout_sec: float,
) -> AttemptResult:
    """
    하나의 (base, root, auth) 조합만 평가한다.
    여기서는 password 후보만 순회한다.

    allow_recovery=True 인 경우:
      - factory reset (1234) 경로
      - default password state가 감지되면 target_password로 변경 진행

    allow_recovery=False 인 경우:
      - 이미 비밀번호가 변경된 제품에 대한 직접 접속
      - 자동 비밀번호 변경 로직 실행 안 함
    """
    best_non_transport: AppError | None = None
    last_transport: AppError | None = None

    for password in password_candidates:
        cli = _build_client(
            base_url=base_url,
            root_path=root_path,
            username=req.username,
            password=password,
            auth_scheme=auth_scheme,
            verify_tls=verify_tls,
            timeout_sec=timeout_sec,
        )

        try:
            _ = cli.read_param_text(PASSWORD_TEST_KEY)
        except AppError as exc:
            if _is_transport_miss(exc):
                last_transport = exc
                break

            best_non_transport = exc
            continue

        try:
            essentials = _read_essentials(cli)
            sys_version = (essentials.get("SYS_VERSION") or "").strip()
            if sys_version:
                return AttemptResult(
                    ok=True,
                    base_url=base_url,
                    root_path=root_path,
                    auth_scheme=auth_scheme,
                    flavor=flavor,
                    sys_version=sys_version,
                    effective_username=req.username,
                    effective_password=password,
                    recovered=False,
                    default_password_state=False,
                    password_changed=False,
                    error=None,
                )

            best_non_transport = AppError(
                kind="parse",
                message="SYS_VERSION missing after full connect",
                detail=str(essentials),
            )
            break

        except AppError as exc:
            if allow_recovery and _is_default_password_state(exc):
                try:
                    recover = try_recover_password(
                        base_url=base_url,
                        root_path=root_path,
                        username=req.username,
                        password_candidates=[password],
                        target_password=target_password,
                        timeout_sec=timeout_sec,
                        verify_tls=verify_tls,
                        auth_scheme=auth_scheme,
                    )

                    essentials2 = _verify_after_password_change(
                        base_url=base_url,
                        root_path=root_path,
                        username=req.username,
                        new_password=target_password,
                        auth_scheme=auth_scheme,
                        verify_tls=verify_tls,
                        timeout_sec=timeout_sec,
                    )

                    return AttemptResult(
                        ok=True,
                        base_url=base_url,
                        root_path=root_path,
                        auth_scheme=auth_scheme,
                        flavor=flavor,
                        sys_version=(essentials2.get("SYS_VERSION") or "").strip() or None,
                        effective_username=req.username,
                        effective_password=recover.changed_to,
                        recovered=True,
                        default_password_state=True,
                        password_changed=True,
                        error=None,
                    )

                except AppError as change_err:
                    best_non_transport = change_err
                    break

            if _is_transport_miss(exc):
                last_transport = exc
                break

            best_non_transport = exc
            continue

        except Exception as exc:
            wrapped = AppError(kind="network", message="candidate connect failed", detail=str(exc))
            if _is_transport_miss(wrapped):
                last_transport = wrapped
                break
            best_non_transport = wrapped
            continue

    err = best_non_transport or last_transport or AppError(
        kind="auth",
        message="all password candidates failed",
        status_code=401,
    )

    return AttemptResult(
        ok=False,
        base_url=base_url,
        root_path=root_path,
        auth_scheme=auth_scheme,
        flavor=flavor,
        effective_username=req.username,
        effective_password=None,
        recovered=False,
        default_password_state=None,
        password_changed=None,
        error=err,
    )


class CameraConnectService:
    def __init__(self, *, settings: AppSettings | None = None) -> None:
        self.settings = settings or AppSettings.load()

    def connect(self, request: Phase1Request) -> Phase1Response:
        return run_phase1(request, settings=self.settings)


def run_phase1(req: Phase1Request, *, settings: AppSettings | None = None) -> Phase1Response:
    settings = settings or AppSettings.load()

    target_password = _resolve_target_password(req, settings)
    sec3_username = _resolve_sec3_username(req, settings)
    sec3_password = _resolve_sec3_password(req, settings)

    password_candidates, allow_recovery = _candidate_passwords(req, settings)

    probe_result = None
    probe_error: AppError | None = None

    try:
        probe_result = probe_camera(
            ip=req.ip,
            port=req.port,
            timeout_sec=float(settings.timeout.connect_sec),
            verify_tls=bool(req.verify_tls),
        )
    except AppError as exc:
        probe_error = exc
        probe_result = None

    # Security 3.0 direct branch
    if probe_result is not None and (probe_result.flavor or "").lower() == "security3":
        try:
            _, essentials = security3_bootstrap(
                target_ip=req.ip,
                base_url=probe_result.base_url,
                root_path=probe_result.root_path,
                sec3_username=sec3_username,
                sec3_password=sec3_password,
                allowed_ip=settings.allowed_ip,
                verify_tls=bool(req.verify_tls),
                timeout_sec=float(settings.timeout.read_sec),
            )

            return Phase1Response(
                ok=True,
                base_url=probe_result.base_url,
                root_path=probe_result.root_path,
                auth_scheme="digest",
                flavor="security3",
                sys_version=(essentials.get("SYS_VERSION") or "").strip() or None,
                effective_username=sec3_username,
                effective_password=sec3_password,
                recovered=True,
                default_password_state=False,
                password_changed=False,
                error=None,
            )
        except AppError as exc:
            return Phase1Response(
                ok=False,
                base_url=probe_result.base_url,
                root_path=probe_result.root_path,
                auth_scheme="digest",
                flavor="security3",
                effective_username=sec3_username,
                effective_password=None,
                error=exc,
            )

    direct_result: AttemptResult | None = None

    # 1) probe 조합 먼저 단독 평가
    if probe_result is not None:
        direct_result = _attempt_candidate_combo(
            req=req,
            base_url=probe_result.base_url,
            root_path=probe_result.root_path,
            auth_scheme=probe_result.auth_scheme,
            flavor=probe_result.flavor or "legacy",
            password_candidates=password_candidates,
            target_password=target_password,
            allow_recovery=allow_recovery,
            verify_tls=bool(req.verify_tls),
            timeout_sec=float(settings.timeout.read_sec),
        )

        if direct_result.ok:
            return direct_result.to_phase1_response()

        # 의미 있는 에러면 바로 반환
        if direct_result.error is not None and not _is_transport_miss(direct_result.error):
            return direct_result.to_phase1_response()

    # 2) probe 실패 또는 transport miss일 때만 fallback
    base_candidates = build_base_candidates(req.ip, req.port)
    root_candidates = build_root_candidates(probe_result.root_path if probe_result else None)
    auth_candidates = _unique_keep_order(
        [
            probe_result.auth_scheme if probe_result else None,
            "digest",
            "basic",
            "none",
        ]
    )

    preferred_base = probe_result.base_url if probe_result else None
    preferred_root = probe_result.root_path if probe_result else None
    preferred_auth = probe_result.auth_scheme if probe_result else None

    best_non_transport: AttemptResult | None = None
    last_transport: AttemptResult | None = None

    for base_url in base_candidates:
        for root_path in root_candidates:
            flavor = "tta" if root_path == "/httpapx/" else "legacy"

            for auth_scheme in auth_candidates:
                if (
                    preferred_base
                    and preferred_root
                    and preferred_auth
                    and base_url == preferred_base
                    and root_path == preferred_root
                    and auth_scheme == preferred_auth
                ):
                    continue

                result = _attempt_candidate_combo(
                    req=req,
                    base_url=base_url,
                    root_path=root_path,
                    auth_scheme=auth_scheme,
                    flavor=flavor,
                    password_candidates=password_candidates,
                    target_password=target_password,
                    allow_recovery=allow_recovery,
                    verify_tls=bool(req.verify_tls),
                    timeout_sec=float(settings.timeout.read_sec),
                )

                if result.ok:
                    return result.to_phase1_response()

                if result.error is not None and not _is_transport_miss(result.error):
                    if best_non_transport is None:
                        best_non_transport = result
                    continue

                last_transport = result

    if best_non_transport is not None:
        return best_non_transport.to_phase1_response()

    if direct_result is not None:
        return direct_result.to_phase1_response()

    if last_transport is not None:
        return last_transport.to_phase1_response()

    return Phase1Response(
        ok=False,
        base_url=(probe_result.base_url if probe_result else None),
        root_path=(probe_result.root_path if probe_result else None),
        auth_scheme=(probe_result.auth_scheme if probe_result else None),
        flavor=(probe_result.flavor if probe_result else "legacy") if probe_result else None,
        effective_username=req.username,
        effective_password=None,
        error=probe_error or AppError(kind="probe", message="probe/connect failed"),
    )
