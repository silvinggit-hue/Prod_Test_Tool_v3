from __future__ import annotations

import os

import pytest

from application.services.connect_service import run_phase1
from config.app_settings import AppSettings
from domain.models.phase1 import Phase1Request


@pytest.mark.skipif(
    not os.environ.get("PTT3_REAL_IP"),
    reason="requires real camera environment variables",
)
def test_real_phase1_sys_version_success() -> None:
    ip = os.environ["PTT3_REAL_IP"]
    port = int(os.environ.get("PTT3_REAL_PORT", "0"))
    username = os.environ.get("PTT3_REAL_USER", "admin")
    password = os.environ.get("PTT3_REAL_PASSWORD", "123")
    verify_tls = os.environ.get("PTT3_REAL_VERIFY_TLS", "false").lower() in ("1", "true", "yes")

    settings = AppSettings.load()

    req = Phase1Request(
        ip=ip,
        port=port,
        username=username,
        password=password,
        password_candidates=("123", "!camera1108"),
        target_password="!camera1108",
        verify_tls=verify_tls,
        sec3_username="TruenTest",
        sec3_password="!Camera1108",
        allowed_ip=settings.allowed_ip,
    )

    resp = run_phase1(req, settings=settings)

    assert resp.ok is True
    assert resp.base_url
    assert resp.root_path
    assert resp.auth_scheme
    assert resp.sys_version
